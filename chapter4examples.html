      function range(start, end, step=1) {
        console.log('step is: ' + step)
        let numArray = [];
        for (i = start; Math.abs(i) != Math.abs(end) + 1; i = i + step) {
            numArray.push(i);
          console.log('i is: ' + i)
          }
        return numArray
        }
      
      function sum(array) {
          let total = 0;
          for (i = 0; i < array.length; i++) {
            let num = array[i];
            total = total + num;
          }
        return total;
        }
